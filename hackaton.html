nano33_gesture.ino
/* GESTURE HERO — Arduino Nano 33 BLE Sense
 * IA embarquée (Edge Impulse) + Serial USB */

#include <Arduino_LSM9DS1.h>
#include <Mr_Sun-project-1_inferencing.h>

#define NUM_SAMPLES       119
#define NUM_AXES          6
#define CONFIDENCE_THRESH 0.70

float features[NUM_SAMPLES * NUM_AXES];
int sampleIndex = 0;
String lastGesture = "";

void setup() {
    Serial.begin(115200);
    while (!Serial);
    if (!IMU.begin()) { Serial.println("IMU ERROR"); while(1); }
    Serial.println("Gesture Hero Ready");
}

void loop() {
    sampleIndex = 0;
    while (sampleIndex < NUM_SAMPLES) {
        float ax, ay, az, gx, gy, gz;
        if (IMU.accelerationAvailable() && IMU.gyroscopeAvailable()) {
            IMU.readAcceleration(ax, ay, az);
            IMU.readGyroscope(gx, gy, gz);
            features[sampleIndex * NUM_AXES + 0] = ax;
            features[sampleIndex * NUM_AXES + 1] = ay;
            features[sampleIndex * NUM_AXES + 2] = az;
            features[sampleIndex * NUM_AXES + 3] = gx;
            features[sampleIndex * NUM_AXES + 4] = gy;
            features[sampleIndex * NUM_AXES + 5] = gz;
            sampleIndex++;
        }
    }
    signal_t signal;
    signal.total_length = NUM_SAMPLES * NUM_AXES;
    signal.get_data = &raw_feature_get_data;
    ei_impulse_result_t result;
    if (run_classifier(&signal, &result, false) == EI_IMPULSE_OK) {
        int maxIdx = 0; float maxVal = 0;
        for (int i = 0; i < EI_CLASSIFIER_LABEL_COUNT; i++) {
            if (result.classification[i].value > maxVal) {
                maxVal = result.classification[i].value; maxIdx = i;
            }
        }
        if (maxVal >= CONFIDENCE_THRESH) {
            String g = result.classification[maxIdx].label;
            if (g != "idle" && g != lastGesture) {
                Serial.println(g);
                lastGesture = g;
            } else if (g == "idle") { lastGesture = ""; }
        }
    }
    delay(80);
}

int raw_feature_get_data(size_t offset, size_t len, float *out) {
    memcpy(out, features + offset, len * sizeof(float));
    return 0;
}
gesture_to_keys.py
"""GESTURE HERO — Python Keyboard Simulator"""

import serial, time, sys
from pynput.keyboard import Key, Controller

SERIAL_PORT = "COM3"    # or /dev/ttyACM0
BAUD_RATE   = 115200
PRESS_DUR  = 0.15

GESTURE_MAP = {
    "forward":  Key.up,
    "backward": Key.down,
    "left":     Key.left,
    "right":    Key.right,
}

def main():
    kb = Controller()
    print("=== Gesture Hero — Arduino Days 2026 ===")
    print("Commandes :")
    for g, k in GESTURE_MAP.items():
        print(f"  {g} → {k}")
    try:
        ser = serial.Serial(SERIAL_PORT, BAUD_RATE, timeout=1)
        time.sleep(2)
        print("Connecté ! En attente de gestes...")
        while True:
            if ser.in_waiting > 0:
                g = ser.readline().decode("utf-8").strip().lower()
                if g in GESTURE_MAP:
                    kb.press(GESTURE_MAP[g])
                    time.sleep(PRESS_DUR)
                    kb.release(GESTURE_MAP[g])
                    print(f"  → {g.upper()}")
    except KeyboardInterrupt:
        print("\nStopped.")
    finally:
        if 'ser' in locals(): ser.close()

if __name__ == "__main__": main()
